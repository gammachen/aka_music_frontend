                    "success": True,
                    "affected_rows": cursor.rowcount
                }

            return {
                "success": False,
                "error": str(e)

import time
from functools import wraps

class PerformanceMonitor:
    def __init__(self):
        self.metrics = {}
    
    def timer(self, name):
        def decorator(func):
            @wraps(func)
            def wrapper(*args, **kwargs):
                start = time.time()
                try:
                    result = func(*args, **kwargs)
                    self.metrics[f"{name}_success"] = self.metrics.get(f"{name}_success", 0) + 1
                    return result
                except Exception as e:
                    self.metrics[f"{name}_error"] = self.metrics.get(f"{name}_error", 0) + 1
                    raise
                finally:
                    duration = time.time() - start
                    self.metrics[f"{name}_duration"] = self.metrics.get(f"{name}_duration", 0) + duration
            return wrapper
        return decorator
