# 时间标准化示例
def normalize_date(text):
    patterns = [
        (r'天宝(\d+)载', r'天宝\1年'),
        (r'开元末', '开元29年')
    ]
    for pat, repl in patterns:
        text = re.sub(pat, repl, text)
    return text
